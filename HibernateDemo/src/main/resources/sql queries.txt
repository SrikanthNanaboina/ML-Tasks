
USE ML_DB
GO

IF OBJECT_ID('ML_DB.dbo.OWNERS', 'U') IS NOT NULL
  DROP TABLE ML_DB.dbo.OWNERS
GO

CREATE TABLE ML_DB.dbo.OWNERS
(
	OWNER_ID INT  NOT NULL, 
	NAME VARCHAR(20) NOT NULL, 
    	PRIMARY KEY (OWNER_ID)
)
GO

IF OBJECT_ID('ML_DB.dbo.VEHICLES', 'U') IS NOT NULL
  DROP TABLE ML_DB.dbo.VEHICLES
GO

CREATE TABLE ML_DB.dbo.VEHICLES
(
	V_ID INT  NOT NULL, 
	NAME VARCHAR(20) NOT NULL,
	OWNER_ID INT NOT NULL,
    	PRIMARY KEY (V_ID),
	FOREIGN KEY (OWNER_ID) REFERENCES dbo.OWNERS(OWNER_ID)
)
GO
